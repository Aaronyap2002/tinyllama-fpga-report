% extreport is the report class from the extsizes package.
% It supports a fixed set of sizes: 8pt 9pt 10pt 11pt 12pt 14pt 17pt 20pt.
% We use 11pt as the closest available size to the NTU template's 10.5pt.
% Unlike scrreprt (KOMA-Script), extreport behaves exactly like the standard
% report class -- all standard packages work with it without conflicts.
\documentclass[11pt, a4paper]{extreport}

% -- Core font (Times New Roman) ----------------------------------------------
% mathptmx switches both the text font and the math font to Times New Roman.
\usepackage{mathptmx}

% -- Page geometry ------------------------------------------------------------
% Left margin is wider (3.5cm) to match the NTU Word template binding margin.
\usepackage[
  a4paper,
  left       = 3.5cm,
  right      = 3.0cm,
  top        = 3.0cm,
  bottom     = 3.0cm,
  headheight = 14pt
]{geometry}

% -- Line spacing -------------------------------------------------------------
% The NTU Word template uses 1.5x line spacing throughout the body text.
% With extreport (unlike scrreprt), calling \onehalfspacing directly in the
% preamble is perfectly safe -- there is no internal geometry recalculation
% that conflicts with setspace's font metric changes.
\usepackage{setspace}
\onehalfspacing

% -- Mathematics --------------------------------------------------------------
\usepackage{amsmath}    % align, cases, split environments
\usepackage{amssymb}    % extra math symbols
\usepackage{bm}         % \bm{W} for bold math (weight matrices, vectors)

% -- Graphics and figures -----------------------------------------------------
\usepackage{graphicx}   % \includegraphics[width=0.8\linewidth]{...}
\usepackage{caption}    % controls caption font and spacing
\usepackage{subcaption} % \begin{subfigure} for side-by-side figures
\usepackage{float}      % [H] placement specifier
\usepackage{makecell}   % line breaks inside table cells

% -- Tables -------------------------------------------------------------------
\usepackage{booktabs}   % \toprule \midrule \bottomrule
\usepackage{array}      % extended column types
\usepackage{multirow}   % cells spanning multiple rows
\usepackage{tabularx}   % tables that stretch to text width

% -- Code listings ------------------------------------------------------------
\usepackage{listings}
\usepackage{xcolor}
\usepackage{inconsolata}  % clean monospace font for code

\definecolor{codekeyword}{RGB}{30, 100, 200}
\definecolor{codecomment}{RGB}{0, 128, 96}

\lstset{
  language         = C++,
  basicstyle       = \ttfamily\small,
  keywordstyle     = \color{codekeyword}\bfseries,
  commentstyle     = \color{codecomment}\itshape,
  morekeywords     = {pragma, HLS, PIPELINE, UNROLL, ARRAY_PARTITION,
                      II, factor, cyclic, complete, dataflow},
  numbers          = left,
  numberstyle      = \tiny\color{gray},
  stepnumber       = 1,
  numbersep        = 8pt,
  frame            = tb,
  rulecolor        = \color{gray!50},
  showstringspaces = false,
  breaklines       = true,
  tabsize          = 2,
  captionpos       = b,
}

% -- Headers and footers (fancyhdr) -------------------------------------------
% fancyhdr works cleanly with extreport because extreport is a standard class.
% The conflict warnings we saw before were specific to using fancyhdr with
% KOMA-Script's scrreprt, which has its own incompatible header/footer system.
\usepackage{fancyhdr}
\pagestyle{fancy}
\fancyhf{}                                  % clear all header/footer fields
\fancyhead[L]{\small\leftmark}              % chapter name on left of header
\fancyfoot[C]{\thepage}                     % page number centred in footer
\renewcommand{\headrulewidth}{0.4pt}        % thin rule below header
\renewcommand{\footrulewidth}{0pt}          % no rule above footer

% Also apply fancy style to chapter opening pages (which default to 'plain')
\fancypagestyle{plain}{
  \fancyhf{}
  \fancyfoot[C]{\thepage}
  \renewcommand{\headrulewidth}{0pt}
}

% -- Chapter and section heading style (titlesec) -----------------------------
% titlesec works cleanly with extreport. The conflicts we saw before were
% specific to titlesec trying to redefine KOMA-Script's internal heading
% commands, which KOMA handles very differently from the standard classes.
\usepackage{titlesec}

% Chapter heading: "Chapter 1  Introduction" left-aligned, 16pt bold.
% The [hang] shape puts the chapter number and title on the same line.
% {before-sep}{after-sep} control vertical spacing around the heading.
\titleformat{\chapter}[hang]
  {\normalfont\fontsize{16pt}{20pt}\selectfont\bfseries}
  {Chapter \thechapter\quad}
  {0pt}
  {}
\titlespacing*{\chapter}{0pt}{30pt}{20pt}

\titleformat{\section}
  {\normalfont\large\bfseries}
  {\thesection}
  {1em}
  {}
\titlespacing*{\section}{0pt}{12pt}{6pt}

\titleformat{\subsection}
  {\normalfont\normalsize\bfseries\itshape}
  {\thesubsection}
  {1em}
  {}
\titlespacing*{\subsection}{0pt}{10pt}{4pt}

% Unnumbered chapters (Abstract, Acknowledgement etc.) use the same font
% but without a chapter number prefix.
\titleformat{name=\chapter,numberless}[block]
  {\normalfont\fontsize{16pt}{20pt}\selectfont\bfseries\centering}
  {}
  {0pt}
  {}
\titlespacing*{name=\chapter,numberless}{0pt}{30pt}{20pt}

% -- Bibliography -------------------------------------------------------------

  style   = ieee,    % numbered citations [1], [2]
  backend = biber,   % biber handles UTF-8 and DOIs correctly
  sorting = none     % references in citation order (required for IEEE)

% \mainmatter, or \backmatter -- those belong to the book class.

  \cleardoublepage
  \pagenumbering{roman}%   % i, ii, iii ... for abstract, TOC, acronyms

  \cleardoublepage
  \pagenumbering{arabic}%  % 1, 2, 3 ... resets counter for chapter content
}
\newcommand{\backmatter}{%
  \cleardoublepage
}
